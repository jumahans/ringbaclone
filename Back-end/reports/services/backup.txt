def submit_microsoft_fraud(
    phone_number: str,
    landing_url: str
) -> Tuple[bool, str]:
    try:
        logger.info(f"Microsoft fraud report queued for {phone_number}")
        return (True, "Microsoft fraud report queued")
    except Exception as e:
        logger.error(f"Microsoft fraud submission failed: {e}")
        return (False, str(e))
